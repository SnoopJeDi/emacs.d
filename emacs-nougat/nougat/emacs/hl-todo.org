** hl-todo
hl-todo hilights words in the active buffer like 'todo' and 'fixme'.
#+begin_src emacs-lisp
(use-package hl-todo
 :straight (hl-todo :type git
               :host github
               :repo "tarsius/hl-todo")
 :config
       (setq global-hl-todo-mode t)
       (setq hl-todo-keyword-faces
             '(("todo"   . "#FF0000")
               ("fixme"  . "#FF0000"))
       )
)
#+end_src
